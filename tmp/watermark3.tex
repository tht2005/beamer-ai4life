\usepackage{tikz}
\usepackage{xcolor}
\usepackage{graphicx}

\definecolor{ai4life}{RGB}{50,130,160}

\begin{document}
\begin{tikzpicture}[x=1cm,y=1cm]

% ------------------------------
% Bounding box (full page)
\useasboundingbox (0,0) rectangle (16,9);

% ------------------------------
% Decorative grid
\foreach \x in {1,3,...,15}
  \foreach \y in {1,3,...,8}
    \fill[ai4life!8] (\x,\y) circle (0.1);

% ------------------------------
% Neural network layers (scaled to fit page)
\foreach \y/\name in {1/m1,2/m2,3/m3,4/p1,5/p2,6/p3}
    \node[circle,minimum size=0.5cm,draw=ai4life!50,fill=ai4life!10] 
    (input-\name) at (1,\y) {};

\foreach \y/\name in {1/m1,2/m2,3/zero,4/p2,5/p3}
    \node[circle,minimum size=0.5cm,draw=ai4life!50,fill=ai4life!5] 
    (hidden1-\name) at (5,\y) {};

\foreach \y/\name in {1/m1,2/m2,3/p2,4/p3}
    \node[circle,minimum size=0.5cm,draw=ai4life!50,fill=ai4life!5] 
    (hidden2-\name) at (10,\y) {};

    \node[circle,minimum size=0.5cm,draw=ai4life!50,fill=ai4life!10] 
    (output-\name) at (15,\y) {};

% ------------------------------
% Connect layers with faint lines
\foreach \a in {m1,m2,m3,p1,p2,p3}
  \foreach \b in {m1,m2,zero,p2,p3}
    \draw[ai4life!20] (input-\a) -- (hidden1-\b);

\foreach \a in {m1,m2,zero,p2,p3}
  \foreach \b in {m1,m2,p2,p3}
    \draw[ai4life!20] (hidden1-\a) -- (hidden2-\b);

\foreach \a in {m1,m2,p2,p3}
  \foreach \b in {m1,zero,p1}
    \draw[ai4life!20] (hidden2-\a) -- (output-\b);

% ------------------------------
% Faded logo overlay
\node[opacity=0.15, rotate=30] at (8,2.5) {
    \includegraphics[width=10cm]{AI4LIFE_logo.png}
};
